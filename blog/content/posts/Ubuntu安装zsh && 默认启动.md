zsh拥有更强的补全功能(常用工具的参数补全), 推荐大家使用

<!-- more -->
# 安装
`sudo apt-get install zsh`

# 引入增强插件,支持git,rails等补全，可选多种外观皮肤
`wget https://github.com/robbyrussell/oh-my-zsh/raw/master/tools/install.sh -O - | sh`

# zsh作为默认shell，重启后生效
chsh -s /bin/zsh
